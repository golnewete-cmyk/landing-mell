import streamlit as st import os import datetime import requests import base64 import xml.etree.ElementTree as ET from fpdf import FPDF # --- SISTEMA DE IMAGENS (BLINDAGEM CLOUD) --- def carregar_imagem_cloud(caminho): try: if os.path.exists(caminho): with open(caminho, "rb") as f: img_data = base64.b64encode(f.read()).decode() return f"data:image/png;base64,{img_data}" return None except: return None # --- CONEXÃO INTELIGENTE (GOOGLE NEWS BYPASS) --- def buscar_google_trends(): try: url = "https://news.google.com/rss?hl=pt-BR&gl=BR&ceid=BR:pt-150" headers = {"User-Agent": "Mozilla/5.0"} resp = requests.get(url, headers=headers, timeout=10) if resp.status_code == 200: root = ET.fromstring(resp.content) trends = [] for item in root.findall(".//item")[:10]: title = item.find("title").text.split(" - ")[0].upper() trends.append(f"● {title}") return "\n".join(trends) return "TENDÊNCIAS DE MERCADO E ESTRATÉGIA DIGITAL" except: return "INTELIGÊNCIA DE DADOS E NETWORKING" # --- MOTOR PDF PREMIUM (MELLEN STYLE) --- class DossieMellen(FPDF): def __init__(self, cliente): super().__init__() self.cliente = cliente.encode('latin-1', 'ignore').decode('latin-1') def header(self): if self.page_no() > 1: self.set_font('Arial', 'B', 8) self.set_text_color(197, 160, 89) self.cell(0, 10, f'INTEL REPORT | {self.cliente.upper()}', 0, 1, 'R') def capa(self): self.add_page() self.set_fill_color(15, 15, 15) self.rect(0, 0, 210, 297, 'F') self.set_y(120) self.set_font('Arial', 'B', 35) self.set_text_color(197, 160, 89) self.cell(0, 20, self.cliente.upper(), ln=True, align='C') self.set_font('Arial', '', 12) self.set_text_color(255, 255, 255) self.cell(0, 10, "REPERTÓRIO QUE MOLDA", ln=True, align='C') def add_secao(self, tit, msg): self.add_page() self.set_fill_color(15, 15, 15);
